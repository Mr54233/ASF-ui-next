**老版本实现** (`Log.vue`):

**老版本 Key 解析** (`BGR/Input.vue`):
```javascript
const keyRegex = /[0-9A-Z]{4,7}-[0-9A-Z]{4,7}-[0-9A-Z]{4,7}(?:(?:-[0-9A-Z]{4,7})?(?:-[0-9A-Z]{4,7}))?/;
const commonDelimiters = [':', ';', '|', '-', ','];

detectKeyNamePair(line) {
  if (!keyRegex.test(line)) return { key: null, name: line };

  const key = keyRegex.exec(line)[0];
  const keyIndex = line.indexOf(key);
  const name = line.replace(key, '').trim();

  // 处理分隔符
  const possibleDelimiter = name.charAt((keyIndex === 0) ? 0 : name.length - 1);
  if (commonDelimiters.includes(possibleDelimiter)) {
    return { key, name: name.slice((keyIndex === 0) ? 1 : 0, name.length - ((keyIndex === 0) ? 0 : 1)).trim() };
  }

  return { key, name };
}
```

```javascript
  const maFile = this.$refs.file.files[0];
  const maFileContent = await this.readFile(maFile);
  const response = await this.$http.post(`bot/${bot}/twoFactorAuthentication`, JSON.parse(maFileContent));
  if (response[bot].Success) {

async refreshToken() {

  if (response[bot].Result && response[bot].Success) {
    this.token = response[bot].Result;
  }
}

// 30 秒后自动刷新 token

    if (this.token === '-----') return;
    await delay(30000);
    this.token = '-----';
  }
}
```
